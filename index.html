function doPost(e) { try { if (!e || !e.postData || !e.postData.contents) { return ContentService.createTextOutput("Error: No data received").setMimeType(ContentService.MimeType.TEXT); } var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. Target the correct tab. // This ensures data goes to "Matches" even if you are looking at the "Passwords" tab. var sheet = ss.getSheetByName("Matches") || ss.getSheets()[0]; var imageUrl = ""; // 2. Handle Image Upload to Drive if (data.image && data.imageName) { try { var folderId = "1idlqO737t6Cg6E4yNpATaKVFUA2gCPO_"; var folder = DriveApp.getFolderById(folderId); var contentType = data.image.split(";")[0].split(":")[1]; var bytes = Utilities.base64Decode(data.image.split(",")[1]); var blob = Utilities.newBlob(bytes, contentType, data.imageName); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); imageUrl = file.getUrl(); } catch (imageError) { // Safety Net: If image fails, log the error but do NOT crash the script imageUrl = "Image Error: " + imageError.toString(); } } // 3. Append Row (A: Timestamp, B: Date, C: Reporter, D: Opponent, E: Winner, F: Loser, G: Score, H: Photo) sheet.appendRow([ new Date(), data.date, data.reporter, data.opponent, data.winner, data.loser, data.score, imageUrl ]); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch (f) { return ContentService.createTextOutput("Error: " + f.toString()).setMimeType(ContentService.MimeType.TEXT); } }
